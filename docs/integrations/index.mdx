## Quick Start

### 1. Wrap with WalletProvider

```tsx
// app/layout.tsx
import { WalletProvider } from '@/contexts';

export default function RootLayout({ children }) {
  return (
    <WalletProvider
      horizonUrl="https://horizon-testnet.stellar.org"
      network="TESTNET"
    >
      {children}
    </WalletProvider>
  );
}
```

### 2. Use Pre-built Components

```tsx
import WalletConnectButton from '@/components/WalletConnectButton';

function Header() {
  return <WalletConnectButton />;
}
```

### 3. Access Wallet State

```tsx
import { useWallet } from '@/contexts';

function Dashboard() {
  const { connected, publicKey, balances } = useWallet();

  if (!connected) return <p>Connect wallet to continue</p>;

  return <p>Balance: {balances[0]?.balance} XLM</p>;
}
```

### 4. Interact with Smart Contracts

```tsx
import { useSorobanContract } from '@/hooks/useSorobanContract';

function TokenBalance({ contractId }) {
  const { callFunction } = useSorobanContract({ contractId });

  const getBalance = async () => {
    const balance = await callFunction('balance', [address]);
    console.log('Token balance:', balance);
  };
}
```
